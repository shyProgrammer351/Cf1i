프로그램시작 
   손님에게 주문 받기

프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter01/FoodOrder.c 

	#include <stdio.h>								 

	 

	// 프로그램시작 

1	int main() 						 

	{ 

		// 변수 선언 및 초기화 

		char mainDish[32] = ""; 

		char sauce[32] = ""; 

2		char drink[32] = ""; 

	 

		// 손님에게 주문 받기 

3		printf( "주문 도와드리겠습니다, 손님.\n" );	 

	 

		// 메인요리 주문 받기 

4		printf( "메인요리는 고르셨습니까: " );		 

		// '케이준치킨샐러드'를 입력한다고 가정	 

5		scanf( "%s", mainDish );			 

	 

		// 소스 주문 받기 

6		printf( "소스는 어떻게 할까요: " );		 

		// '허니머스터드드레싱'을 입력한다고 가정	 

7		scanf( "%s", sauce );				 

	 

		// 음료 주문 받기 

8		printf( "음료는 무엇으로 준비할까요: " );	 

		// '콜라'를 입력한다고 가정	 

9		scanf( "%s", drink );				 

	 

		// 주문 내역 확인 

10		printf( "메인요리는 %s, 소스는 %s, 음료는 %s를 선택하셨습니다. \n", 

		         mainDish, sauce, drink );				 

11		printf( "준비하겠습니다, 잠시만 기다려 주십시오." ); 

	 

		// 프로그램 종료 

12		return 0; 

	}
